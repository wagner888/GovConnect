        // Fun√ß√£o para trocar de contato
        function switchContact(contactId) {
            currentContact = contactId;
            
            // Atualizar contato ativo
            document.querySelectorAll('.contact-item').forEach(item => {
                item.classList.remove('active');
            });
            document.querySelector(`[data-contact="${contactId}"]`).classList.add('active');
            
            // Atualizar header do chat
            const contact = contactsInfo[contactId];
            currentChatName.textContent = contact.name;
            currentChatStatus.textContent = contact.status;
            currentChatStatus.style.color = contact.status.includes('Online') ? '#3ecf8e' : 
                                           contact.status.includes('Ausente') ? '#f4a100' : 
                                           contact.status.includes('Ocupado') ? '#ff4757' : '#888';
            document.querySelector('.chat-avatar').textContent = contact.avatar;
            
            // Fechar menu mobile se aberto
            if (window.innerWidth <= 768) {
                sidebar.classList.remove('active');
            }
            
            // Renderizar mensagens
            renderMessages(contactId);
        }

        searchInput.addEventListener('input', function() {
            const searchTerm = this.value.toLowerCase();
            const contactItems = document.querySelectorAll('.contact-item');
            
            contactItems.forEach(item => {
                const name = item.querySelector('.contact-info h4').textContent.toLowerCase();
                const role = item.querySelector('.contact-info p').textContent.toLowerCase();
                
                if (name.includes(searchTerm) || role.includes(searchTerm)) {

                } else {

            });
