        // Setup event listeners
        function setupEventListeners() {
            // Tab switching
            const tabBtns = document.querySelectorAll('.tab-btn');
            tabBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    // Remove active class from all tabs
                    tabBtns.forEach(b => b.classList.remove('active'));
                    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
                    
                    // Add active class to clicked tab
                    this.classList.add('active');
                    const tabId = this.getAttribute('data-tab');
                    document.getElementById(tabId).classList.add('active');
                });
            });
            
            // Search filter
            const searchInput = document.getElementById('searchInput');
            searchInput.addEventListener('input', filterRobots);
            
            // Status filter
            const statusFilter = document.getElementById('statusFilter');
            statusFilter.addEventListener('change', filterRobots);
            
            // Type filter
            const typeFilter = document.getElementById('typeFilter');
            typeFilter.addEventListener('change', filterRobots);
        }

        // Filter robots based on search and filters
        function filterRobots() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            const statusFilter = document.getElementById('statusFilter').value;
            const typeFilter = document.getElementById('typeFilter').value;
            
            const rows = document.querySelectorAll('#robotsTableBody tr');
            
            rows.forEach(row => {
                const name = row.cells[0].textContent.toLowerCase();
                const type = row.cells[1].textContent.toLowerCase();
                const status = row.cells[2].textContent.toLowerCase();
                
                let show = true;
                
                // Search filter
                if (searchTerm && !name.includes(searchTerm)) {
                    show = false;
                }
                
                // Status filter
                if (statusFilter) {
                    let statusMatch = false;
                    if (statusFilter === 'active' && status.includes('ativo')) statusMatch = true;
                    if (statusFilter === 'paused' && status.includes('pausado')) statusMatch = true;
                    if (statusFilter === 'error' && status.includes('erro')) statusMatch = true;
                    if (!statusMatch) show = false;
                }
                
                // Type filter
                if (typeFilter && !type.includes(typeFilter)) {
                    show = false;
                }
                
                row.style.display = show ? '' : 'none';
            });
        }

        // Clear filters
        function clearFilters() {
            document.getElementById('searchInput').value = '';
            document.getElementById('statusFilter').value = '';
            document.getElementById('typeFilter').value = '';
            filterRobots();
        }

        // View robot details
        function viewRobot(id) {
            const robot = robotsData.find(r => r.id === id);
            if (!robot) return;
            
            // Open modal with robot details
            const modalContent = document.getElementById('modalContent');
            
            let typeIcon = 'fa-search';
            let typeClass = 'scanner';
            if (robot.type === 'automation') {
                typeIcon = 'fa-cogs';
                typeClass = 'automation';
            } else if (robot.type === 'collector') {
                typeIcon = 'fa-database';
                typeClass = 'collector';
            }
            
            let statusText = 'Ativo';
            let statusClass = 'status-active';
            if (robot.status === 'paused') {
                statusText = 'Pausado';
                statusClass = 'status-paused';
            } else if (robot.status === 'error') {
                statusText = 'Erro';
                statusClass = 'status-error';
            }
            
            modalContent.innerHTML = `
                <div class="robot-info-card">
                    <div class="robot-header">
                        <div class="robot-avatar">
                            <i class="fas ${typeIcon}"></i>
                        </div>
                        <div class="robot-title">
                            <div class="robot-name">${robot.name}</div>
                            <div class="robot-subtitle">${robot.type.charAt(0).toUpperCase() + robot.type.slice(1)} • ID: ROB-00${robot.id}</div>
                        </div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Descrição</div>
                        <div class="info-value">${robot.description}</div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Status</div>
                        <div class="info-value">
                            <span class="status-badge ${statusClass}">${statusText}</span>
                        </div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Última Execução</div>
                        <div class="info-value">${robot.lastRun}</div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Próxima Execução</div>
                        <div class="info-value">${robot.nextRun}</div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Tentativas Totais</div>
                        <div class="info-value">${robot.attempts.toLocaleString()}</div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Taxa de Sucesso</div>
                        <div class="info-value">${robot.successRate}%</div>
                    </div>

                    <div class="info-item">
                        <div class="info-label">Criado em</div>
                        <div class="info-value">${robot.createdAt}</div>
                    </div>

                        <div class="info-label">Responsável</div>
                        <div class="info-value">${robot.createdBy}</div>
                    </div>
                </div>

                <div class="logs-section">
                    <div class="logs-header">
                        <h3 class="form-section-title">
                            <i class="fas fa-clipboard-list"></i>
                            Logs de Execução Recentes
