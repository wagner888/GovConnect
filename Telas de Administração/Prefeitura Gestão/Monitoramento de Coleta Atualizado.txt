        function formatTime(seconds) {
            if (seconds === 0) return '--';
            const mins = Math.floor(seconds / 60);
            const secs = seconds % 60;
            return `${mins}m ${secs}s`;
        }

        function formatVolume(bytes) {
            if (bytes === 0) return '--';
            if (bytes < 1024) return bytes + ' MB';
            return (bytes / 1024).toFixed(1) + ' GB';
        }

        function renderTable() {

        function filterData() {
            const estadoFilter = document.getElementById('estadoFilter').value;
            const ibgeFilter = document.getElementById('ibgeFilter').value.toLowerCase();
            const statusFilter = document.getElementById('statusFilter').value;
            const fontesFilter = document.getElementById('fontesFilter').value;
            const buscarFonte = document.getElementById('buscarFonte').value.toLowerCase();
            const tableStatusFilter = document.getElementById('tableStatusFilter').value;
            const tableSearch = document.getElementById('tableSearch').value.toLowerCase();

            return tableData.filter(item => {
                return (
                    (estadoFilter === '' || true) &&
                    (ibgeFilter === '' || item.ibge.toLowerCase().includes(ibgeFilter)) &&
                    (statusFilter === '' || item.status.toLowerCase() === statusFilter) &&
                    (fontesFilter === '' || item.fonte.toLowerCase().includes(fontesFilter)) &&
                    (buscarFonte === '' || item.fonte.toLowerCase().includes(buscarFonte)) &&
                    (tableStatusFilter === '' || item.status === tableStatusFilter) &&
                    (tableSearch === '' ||
                     item.ibge.toLowerCase().includes(tableSearch) ||
                     item.cidade.toLowerCase().includes(tableSearch) ||
                     item.fonte.toLowerCase().includes(tableSearch) ||
                     item.status.toLowerCase().includes(tableSearch))
                );
            });
        }

        function sortData(data) {
            return [...data].sort((a, b) => {
                let valueA = a[currentSort.column];
                let valueB = b[currentSort.column];

                if (currentSort.column === 'data') {
                    valueA = new Date(valueA);
                    valueB = new Date(valueB);
                }

                if (typeof valueA === 'string') {
                    return currentSort.direction === 'asc' 
                        ? valueA.localeCompare(valueB)
                        : valueB.localeCompare(valueA);
                } else {
                    return currentSort.direction === 'asc'
                        ? valueA - valueB
                        : valueB - valueA;
                }
            });
        }

        function updatePaginationInfo(totalItems) {
            const totalPages = Math.ceil(totalItems / rowsPerPage);
            document.getElementById('pageInfo').textContent = `Página ${currentPage} de ${totalPages}`;
            document.getElementById('prevPage').disabled = currentPage === 1;
            document.getElementById('nextPage').disabled = currentPage === totalPages || totalPages === 0;
        }

                    const currentOrder = th.getAttribute('data-order');
                    
                    // Reset all sort indicators
                    document.querySelectorAll('th[data-sort]').forEach(t => {
                        t.setAttribute('data-order', '');
                        t.querySelector('i').className = 'fas fa-sort';
                    });
                    
                    // Set new sort order
                    currentSort.column = column;
                    currentSort.direction = currentOrder === 'asc' ? 'desc' : 'asc';
                    th.setAttribute('data-order', currentSort.direction);
                    th.querySelector('i').className = `fas fa-sort-${currentSort.direction === 'asc' ? 'up' : 'down'}`;
                    
                    renderTable();
                });
            });
        }

                    renderTable();
                });
            });
            
            // Paginação
            document.getElementById('prevPage').addEventListener('click', () => {
                if (currentPage > 1) {
                    currentPage--;
                    renderTable();
                }
            });
            
            document.getElementById('nextPage').addEventListener('click', () => {
                const filteredData = filterData();
                const totalPages = Math.ceil(filteredData.length / rowsPerPage);
                if (currentPage < totalPages) {
                    currentPage++;
                    renderTable();
                }
            });
        });
