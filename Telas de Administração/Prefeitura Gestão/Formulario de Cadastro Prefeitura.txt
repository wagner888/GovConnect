        function refreshPage() {
            const btn = document.querySelector('.btn[onclick="refreshPage()"]');
            const originalText = btn.innerHTML;
            
            btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Atualizando...';
            btn.disabled = true;
            
            setTimeout(() => {
                btn.innerHTML = originalText;
                btn.disabled = false;
                
                const notification = document.querySelector('.notification-badge');
                notification.style.animation = 'none';
                notification.offsetHeight;
                notification.style.animation = null;
            }, 1500);
        }

        $(document).ready(function() {
            // Máscara dinâmica para telefone
            var behavior = function (val) {
                return val.replace(/\D/g, '').length === 11 ? '(00) 00000-0000' : '(00) 0000-00009';
            },
            options = {
                onKeyPress: function(val, e, field, options) {
                    field.mask(behavior.apply({}, arguments), options);
                }
            };

            // Aplicar máscaras fixas
            $('#cnpj').mask('00.000.000/0000-00');
            $('#cpfContato').mask('000.000.000-00');
            $('#cpfContatoFinanceiro').mask('000.000.000-00');
            $('#whatsapp').mask('(00) 00000-0000');
            $('#whatsappFinanceiro').mask('(00) 00000-0000');
            $('#cep').mask('00000-000');
            $('#telefone1').mask('(00) 0000-0000');
            $('#telefone2').mask('(00) 0000-0000');
            $('#telefone1Financeiro').mask('(00) 0000-0000');
            $('#telefone2Financeiro').mask('(00) 0000-0000');

            // Calcular período de vigência
            $('#dataContrato, #vencimentoContrato').change(function() {
                var inicio = new Date($('#dataContrato').val());
                var fim = new Date($('#vencimentoContrato').val());
                if (!isNaN(inicio) && !isNaN(fim)) {
                    var diff = Math.floor((fim - inicio) / (1000 * 60 * 60 * 24));
                    $('#periodoVigencia').val(diff + ' dias');
                }
            });

            // Validação do formulário
            $('#prefeituraForm').on('submit', function(e) {
                e.preventDefault();
                // Aqui você pode adicionar validações adicionais
                alert('Formulário enviado com sucesso!');
                // this.submit(); // Descomente para enviar realmente o formulário
            });

            // Alternar modo de edição
            $('#btnEditar').click(function() {
                $('input, select').prop('disabled', false);
                alert('Modo de edição ativado!');
            });
            
            // Adicionar evento para o botão Sair
            document.querySelector('.logout-icon').addEventListener('click', () => {
                if (confirm('Tem certeza que deseja sair do sistema?')) {
                    showToast('Saindo do sistema...', 'info');
                    setTimeout(() => {
                        window.location.href = 'login.html';
                    }, 1500);
                }
            });
            
            // Função para mostrar toast
            function showToast(message, type = 'info') {
                const toast = document.createElement('div');
                toast.style.cssText = `
                    position: fixed;
                    top: 100px;
                    right: 24px;
                    background: ${type === 'success' ? '#3ecf8e' : '#fbbf24'};
                    color: white;
                    padding: 12px 18px;
                    border-radius: 25px;
                    font-size: 0.9rem;
                    box-shadow: 0 4px 15px rgba(0,0,0,0.15);
                    z-index: 1000;
                    opacity: 0;
                    transform: translateY(-20px);
                    transition: all 0.3s ease;
                    display: flex;
                    align-items: center;
                    gap: 8px;
                    border: 1px solid rgba(255,255,255,0.2);
                `;
                
                toast.innerHTML = `
                    <i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i>
                    ${message}
                `;
                
                document.body.appendChild(toast);
                
                setTimeout(() => {
                    toast.style.opacity = '1';
                    toast.style.transform = 'translateY(0)';
                }, 100);
                
                setTimeout(() => {
                    toast.style.opacity = '0';
                    toast.style.transform = 'translateY(-20px)';
                    setTimeout(() => toast.remove(), 300);
                }, 3000);
            }
        });
