        function refreshPage() {
            const btn = document.querySelector('.btn[onclick="refreshPage()"]');
            const originalText = btn.innerHTML;
            
            btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Atualizando...';
            btn.disabled = true;
            
            setTimeout(() => {
                btn.innerHTML = originalText;
                btn.disabled = false;
                
                const notification = document.querySelector('.notification-badge');
                notification.style.animation = 'none';
                notification.offsetHeight;
                notification.style.animation = null;
                
                console.log('Página atualizada');
            }, 1500);
        }

        function showDistrictFederalMessage() {
            alert('Distrito Federal não possui prefeituras municipais. O DF é uma unidade federativa composta apenas pelo governo distrital.');
        }

        // Adiciona animação de entrada para os cartões
        document.addEventListener('DOMContentLoaded', () => {
            const stateCards = document.querySelectorAll('.state-card');
            stateCards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'opacity 0.5s ease, transform 0.5s ease';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 50);
            });
            
            // Adicionar evento para o botão Sair
            document.querySelector('.logout-icon').addEventListener('click', () => {
                if (confirm('Tem certeza que deseja sair do sistema?')) {
                    showToast('Saindo do sistema...', 'info');
                    setTimeout(() => {
                        window.location.href = 'login.html';
                    }, 1500);
                }
            });
            
            // Função para mostrar toast
            function showToast(message, type = 'info') {
                const toast = document.createElement('div');
                toast.style.cssText = `
                    position: fixed;
                    top: 100px;
                    right: 24px;
                    background: ${type === 'success' ? '#3ecf8e' : '#fbbf24'};
                    color: white;
                    padding: 12px 18px;
                    border-radius: 25px;
                    font-size: 0.9rem;
                    box-shadow: 0 4px 15px rgba(0,0,0,0.15);
                    z-index: 1000;
                    opacity: 0;
                    transform: translateY(-20px);
                    transition: all 0.3s ease;
                    display: flex;
                    align-items: center;
                    gap: 8px;
                    border: 1px solid rgba(255,255,255,0.2);
                `;
                
                toast.innerHTML = `
                    <i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i>
                    ${message}
                `;
                
                document.body.appendChild(toast);
                
                setTimeout(() => {
                    toast.style.opacity = '1';
                    toast.style.transform = 'translateY(0)';
                }, 100);
                
                setTimeout(() => {
                    toast.style.opacity = '0';
                    toast.style.transform = 'translateY(-20px)';
                    setTimeout(() => toast.remove(), 300);
                }, 3000);
            }
        });
    </script>
