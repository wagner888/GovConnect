        // Dados simulados
        const STATES = ["running", "idle", "completed", "error"];
        function randomUrl() {
            const bases = [
                "https://loja.example.com/produtos",
                "https://api.parceiro.com/list",
                "https://sistema.interno.local/relatorios",
                "https://search.example.com/query"
            ];
            const base = bases[Math.floor(Math.random() * bases.length)];
            const q = Math.floor(Math.random() * 999);
            return base + "?job=" + q;
        }
        function createWorker(id) {
            const state = STATES[Math.floor(Math.random() * STATES.length)];
            const runs = Math.floor(Math.random() * 30);
            const secondsAgo = Math.floor(Math.random() * 3600);
            const now = Date.now() - secondsAgo * 1000;
            const duration = Math.floor(Math.random() * 120) + 5;
            return {
                id: "W" + id.toString().padStart(3, "0"),
                state,
                url: randomUrl(),
                runs,
                updatedAt: now,
                lastDurationSec: duration,
                logs: [`[${new Date(now).toISOString()}] Worker inicializado.`]
            };
        }
        const allWorkers = [];
        const TOTAL_WORKERS = 170;
        for (let i = 1; i <= TOTAL_WORKERS; i++) {
            allWorkers.push(createWorker(i));
        }

        // Estado da UI
        let currentFilterState = "all";
        let searchTerm = "";
        let currentPage = 1;
        let pageSize = 10;
        let sortBy = "state";
        let sortDirection = "desc";
        let selectedWorkerId = null;

        // Utilitários
        function formatTimeAgo(ts) {
            const diffSec = Math.floor((Date.now() - ts) / 1000);
            if (diffSec < 60) return diffSec + "s atrás";
            const diffMin = Math.floor(diffSec / 60);
            if (diffMin < 60) return diffMin + "min atrás";
            const diffH = Math.floor(diffMin / 60);
            return diffH + "h atrás";
        }
        function stateLabel(state) {
            switch (state) {
                case "running": return "EM EXECUÇÃO";
                case "idle": return "OCIOSO";
                case "completed": return "CONCLUÍDO";
                case "error": return "ERRO";
                default: return state.toUpperCase();
            }
        }
        function getStatusClass(state) {
            switch (state) {
                case "running": return "status-success";
                case "idle": return "status-warning";
                case "completed": return "status-info";
                case "error": return "status-error";
                default: return "status-warning";
            }
        }
        function sliceForPage(items) {
            const start = (currentPage - 1) * pageSize;
            return items.slice(start, start + pageSize);
        }
        function statusSortWeight(state) {
            switch (state) {
                case "running": return 1;
                case "completed": return 2;
                case "idle": return 3;
                case "error": return 4;
                default: return 99;
            }
        }

        // Filtro e ordenação
        function getFilteredWorkers() {
            let data = [...allWorkers];
            if (currentFilterState !== "all") {
                data = data.filter(w => w.state === currentFilterState);
            }
            if (searchTerm.trim() !== "") {
                const term = searchTerm.toLowerCase();
                data = data.filter(w =>
                    w.id.toLowerCase().includes(term) ||
                    w.url.toLowerCase().includes(term) ||
                    w.state.toLowerCase().includes(term)
                );
            }
            data.sort((a, b) => {
                let valueA, valueB;
                if (sortBy === "state") {
                    valueA = statusSortWeight(a.state);
                    valueB = statusSortWeight(b.state);
                } else if (sortBy === "updated") {
                    valueA = a.updatedAt;
                    valueB = b.updatedAt;
                } else if (sortBy === "id") {
                    valueA = a.id;
                    valueB = b.id;
                } else if (sortBy === "runs") {
                    valueA = a.runs;
                    valueB = b.runs;
                } else if (sortBy === "url") {
                    valueA = a.url;
                    valueB = b.url;
                }
                if (typeof valueA === 'string') {
                    return sortDirection === 'asc' ? valueA.localeCompare(valueB) : valueB.localeCompare(valueA);
                } else {
                    return sortDirection === 'asc' ? valueA - valueB : valueB - valueA;
                }
            });
            return data;
        }

        function renderTable() {
            const filtered = getFilteredWorkers();
            const total = filtered.length;
            const totalPages = Math.max(1, Math.ceil(total / pageSize));
            if (currentPage > totalPages) currentPage = totalPages;
            const pageItems = sliceForPage(filtered);

            tbody.innerHTML = "";
            pageItems.forEach(w => {
                const tr = document.createElement("tr");
                tr.dataset.id = w.id;
                tr.innerHTML = `
                    <td>${w.id}</td>
                    <td><span class="status-badge ${getStatusClass(w.state)}">${stateLabel(w.state)}</span></td>
                    <td title="${w.url}">${truncateText(w.url, 35)}</td>
                    <td>${w.runs}</td>
                    <td>${formatTimeAgo(w.updatedAt)}</td>
                    <td>
                        <button class="btn-action" data-action="inspect" data-id="${w.id}">
                            <i class="fas fa-eye"></i> Ver
                        </button>
                    </td>
                `;
                tbody.appendChild(tr);
            });

            tableInfoEl.textContent = `${total} registro${total !== 1 ? 's' : ''}`;
            pageInfoEl.textContent = `Página ${currentPage} de ${totalPages}`;
            document.getElementById('prevPage').disabled = currentPage === 1;
            document.getElementById('nextPage').disabled = currentPage === totalPages;
            updateSortIcons();
        }

        function truncateText(text, max) {
            return text.length <= max ? text : text.slice(0, max - 3) + "...";
        }

        function updateSortIcons() {
            document.querySelectorAll('#workers-table th').forEach(th => {
                const icon = th.querySelector('i');
                const column = th.getAttribute('data-sort');
                if (column === sortBy) {
                    icon.className = `fas fa-sort-${sortDirection === 'asc' ? 'up' : 'down'}`;
                } else {
                    icon.className = 'fas fa-sort';
                }
            });
        }

        function renderCountsAndMetrics() {
            const total = allWorkers.length;
            const running = allWorkers.filter(w => w.state === "running").length;
            const idle = allWorkers.filter(w => w.state === "idle").length;
            const completed = allWorkers.filter(w => w.state === "completed").length;
            const error = allWorkers.filter(w => w.state === "error").length;
            
            // Calcular taxa de uso (robôs ativos / total * 100)
            const usageRate = total > 0 ? Math.round((running / total) * 100) : 0;

            countAllEl.textContent = total;
            countRunningEl.textContent = running;
            countIdleEl.textContent = idle;
            countCompletedEl.textContent = completed;
            countErrorEl.textContent = error;

            metricActiveEl.textContent = running;
            metricFailEl.textContent = error;
            metricCompletedEl.textContent = completed;
            metricUsageEl.textContent = usageRate + "%";

            // Atualizar barras de progresso nos cards
            const maxActive = Math.max(total, 1); // Evitar divisão por zero
            progressActiveEl.style.width = (running / maxActive * 100) + "%";
            progressFailEl.style.width = (error / maxActive * 100) + "%";
            progressCompletedEl.style.width = (completed / maxActive * 100) + "%";
            progressUsageEl.style.width = usageRate + "%";

            metricRunningEl.textContent = running;
            metricIdleEl.textContent = idle;
            metricErrorEl.textContent = error;

            const pct = x => total === 0 ? 0 : (x / total) * 100;
            barRunningEl.style.width = pct(running) + "%";
            barIdleEl.style.width = pct(idle) + "%";
            barErrorEl.style.width = pct(error) + "%";
        }

        function renderDetail(worker) {
            if (!worker) {
                detailIdTagEl.textContent = "--";
                detailTitleEl.textContent = "Nenhum selecionado";
                detailStatusEl.textContent = "--";
                detailRunsEl.textContent = "--";
                detailUpdatedEl.textContent = "--";
                detailDurationEl.textContent = "--";
                detailUrlEl.textContent = "--";
                detailLogsEl.textContent = "Nenhum log disponível.";
                return;
            }
            selectedWorkerId = worker.id;
            detailIdTagEl.textContent = worker.id;
            detailTitleEl.textContent = `Robô ${worker.id}`;
            detailStatusEl.innerHTML = `<span class="status-badge ${getStatusClass(worker.state)}">${stateLabel(worker.state)}</span>`;
            detailRunsEl.textContent = worker.runs + " execução(ões)";
            detailUpdatedEl.textContent = formatTimeAgo(worker.updatedAt);
            detailDurationEl.textContent = worker.lastDurationSec + "s";
            detailUrlEl.textContent = worker.url;
            detailLogsEl.textContent = worker.logs.join("\n");
        }

        function showToast(message, type = 'info') {
            const toast = document.createElement('div');
            toast.style.cssText = `
                position: fixed; top: 100px; right: 24px; background: ${type === 'success' ? '#3ecf8e' : '#fbbf24'};
                color: white; padding: 12px 18px; border-radius: 25px; font-size: 0.9rem;
                box-shadow: 0 4px 15px rgba(0,0,0,0.15); z-index: 1000; opacity: 0;
                transform: translateY(-20px); transition: all 0.3s ease; display: flex;
                align-items: center; gap: 8px; border: 1px solid rgba(255,255,255,0.2);
            `;
            toast.innerHTML = `<i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i> ${message}`;
            document.body.appendChild(toast);
            setTimeout(() => { toast.style.opacity = '1'; toast.style.transform = 'translateY(0)'; }, 100);
            setTimeout(() => {
                toast.style.opacity = '0';
                toast.style.transform = 'translateY(-20px)';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }

        document.getElementById("btn-restart").addEventListener("click", () => {
            if (!selectedWorkerId) return showToast("Selecione um robô primeiro.", "info");
            const w = allWorkers.find(x => x.id === selectedWorkerId);
            if (!w) return;
            w.state = "running";
            w.runs += 1;
            w.updatedAt = Date.now();
            w.lastDurationSec = 0;
            w.logs.push(`[${new Date().toISOString()}] Reinício manual solicitado pelo operador.`);
            renderCountsAndMetrics();
            renderTable();
            renderDetail(w);
            showToast(`Robô ${w.id} reiniciado com sucesso.`, "success");
        });

        document.getElementById("btn-stop").addEventListener("click", () => {
            if (!selectedWorkerId) return showToast("Selecione um robô primeiro.", "info");
            const w = allWorkers.find(x => x.id === selectedWorkerId);
            if (!w) return;
            w.state = "idle";
            w.updatedAt = Date.now();
            w.logs.push(`[${new Date().toISOString()}] Robô pausado manualmente.`);

        document.getElementById("btn-simulate-start").addEventListener("click", () => {
            const idleWorkers = allWorkers.filter(w => w.state === "idle");
            const toStart = idleWorkers.slice(0, 5);
            if (toStart.length === 0) return showToast("Não há robôs ociosos para iniciar.", "info");
            toStart.forEach(w => {
                w.state = "running";
                w.updatedAt = Date.now();
                w.logs.push(`[${new Date().toISOString()}] Execução simulada iniciada.`);
            });
            renderCountsAndMetrics();
            renderTable();
            if (selectedWorkerId) {
                const w = allWorkers.find(x => x.id === selectedWorkerId);
                if (w) renderDetail(w);
            }
            showToast(`Iniciados ${toStart.length} robôs.`, "success");
        });

        function simulateWorkerTick() {
            allWorkers.forEach(w => {
                const r = Math.random();
                if (w.state === "running") {
                    w.lastDurationSec += 5;
                    if (r < 0.08) {
                        w.state = "completed";
                        w.runs += 1;
                        w.updatedAt = Date.now();
                        w.logs.push(`[${new Date().toISOString()}] Execução concluída com sucesso.`);
                    } else if (r < 0.12) {
                        w.state = "error";
                        w.updatedAt = Date.now();
                        w.logs.push(`[${new Date().toISOString()}] Falha na execução.`);
                    }
                } else if (w.state === "idle" && r < 0.02) {
                    w.state = "running";
                    w.updatedAt = Date.now();
                    w.logs.push(`[${new Date().toISOString()}] Execução automática iniciada.`);
                } else if (w.state === "error" && r < 0.02) {
                    w.state = "idle";
                    w.updatedAt = Date.now();
                    w.logs.push(`[${new Date().toISOString()}] Robô retomado após erro.`);
                } else if (w.state === "completed" && r < 0.01) {
                    w.state = "idle";
                    w.updatedAt = Date.now();
                    w.logs.push(`[${new Date().toISOString()}] Aguardando próxima execução.`);
                }
            });
            renderCountsAndMetrics();
            renderTable();
            if (selectedWorkerId) {
                const w = allWorkers.find(x => x.id === selectedWorkerId);
                if (w) renderDetail(w);
            }
        }

        function atualizarDados() {
            const buttons = document.querySelectorAll('.btn[onclick="atualizarDados()"]');
            buttons.forEach(btn => {
                const original = btn.innerHTML;
                btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Atualizando...';
                btn.disabled = true;
                setTimeout(() => {
                    btn.innerHTML = original;
                    btn.disabled = false;
                }, 1500);
            });
            showToast('Dados atualizados com sucesso!', 'success');
        }

        function init() {
            renderCountsAndMetrics();
            renderTable();
            renderDetail(null);
            setInterval(simulateWorkerTick, 5000);
        }
